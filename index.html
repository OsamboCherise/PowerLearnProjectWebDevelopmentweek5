Part 1: Variables and Conditionals

// Variable declarations
let userName = "Alice";
const userAge = 28;
var isStudent = true;
let userScore = 85;

// Conditional statements
if (userScore >= 90) {
  console.log("Excellent!");
} else if (userScore >= 70) {
  console.log("Good job!");
} else {
  console.log("Keep practicing!");
}

// Switch statement
let grade;
switch(true) {
  case (userScore >= 90):
    grade = "A";
    break;
  case (userScore >= 80):
    grade = "B";
    break;
  case (userScore >= 70):
    grade = "C";
    break;
  default:
    grade = "D";
}

Part 2: Functions

// Function declaration
function calculateArea(width, height) {
  return width * height;
}

// Function expression
const greetUser = function(name) {
  return `Hello, ${name}! Welcome to our website.`;
};

// Arrow function
const multiply = (a, b) => a * b;

// Function with default parameters
function createMessage(name, message = "Hello") {
  return `${message}, ${name}!`;
}

Part 3: Loops

// For loop
for (let i = 1; i <= 5; i++) {
  console.log(`Count: ${i}`);
}

// While loop
let count = 1;
while (count <= 5) {
  console.log(`While count: ${count}`);
  count++;
}

// Loop through an array
const fruits = ["Apple", "Banana", "Orange", "Mango"];
for (let i = 0; i < fruits.length; i++) {
  console.log(fruits[i]);
}

// forEach method (modern approach)
fruits.forEach(fruit => {
  console.log(fruit);
});

Part 4: DOM Interactions

// Changing text content
document.getElementById("myElement").textContent = "New text";

// Changing styles
document.getElementById("myElement").style.color = "blue";

// Adding and removing classes
document.getElementById("myElement").classList.add("highlight");
document.getElementById("myElement").classList.remove("old-class");

// Handling events
document.getElementById("myButton").addEventListener("click", function() {
  alert("Button clicked!");
});

// Creating new elements
const newElement = document.createElement("div");
newElement.textContent = "I'm a new element!";
document.body.appendChild(newElement);
